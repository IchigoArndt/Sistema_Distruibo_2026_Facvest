import { useState } from "react";
import { Card } from "./ui/card";
import { Badge } from "./ui/badge";
import { Tabs, TabsContent, TabsList, TabsTrigger } from "./ui/tabs";
import { Dialog, DialogContent, DialogHeader, DialogTitle } from "./ui/dialog";
import { mockAssessments, type Assessment } from "../data/mockData";
import { Calendar, Clock, CheckCircle2, AlertCircle, Activity } from "lucide-react";

export function Assessments() {
  const [selectedAssessment, setSelectedAssessment] = useState<Assessment | null>(null);

  const requestedAssessments = mockAssessments.filter((a) => a.status === "requested");
  const inProgressAssessments = mockAssessments.filter((a) => a.status === "in-progress");
  const completedAssessments = mockAssessments.filter((a) => a.status === "completed");

  const getStatusBadge = (status: Assessment["status"]) => {
    switch (status) {
      case "requested":
        return (
          <Badge variant="secondary" className="bg-yellow-100 text-yellow-800">
            <Clock className="h-3 w-3 mr-1" />
            Solicitada
          </Badge>
        );
      case "in-progress":
        return (
          <Badge variant="secondary" className="bg-blue-100 text-blue-800">
            <AlertCircle className="h-3 w-3 mr-1" />
            Em Andamento
          </Badge>
        );
      case "completed":
        return (
          <Badge variant="secondary" className="bg-green-100 text-green-800">
            <CheckCircle2 className="h-3 w-3 mr-1" />
            Concluída
          </Badge>
        );
    }
  };

  const formatDate = (dateString: string) => {
    const date = new Date(dateString);
    return date.toLocaleDateString("pt-BR", {
      day: "2-digit",
      month: "long",
      year: "numeric"
    });
  };

  const AssessmentCard = ({ assessment }: { assessment: Assessment }) => (
    <Card
      className="p-5 hover:shadow-lg transition-shadow cursor-pointer border-red-100"
      onClick={() => setSelectedAssessment(assessment)}
    >
      <div className="flex items-start justify-between mb-3">
        <div className="flex-1">
          <h3 className="text-gray-900 mb-1">{assessment.professionalName}</h3>
          <div className="flex items-center space-x-2 text-gray-600 text-sm">
            <Calendar className="h-4 w-4" />
            <span>{formatDate(assessment.date)}</span>
          </div>
        </div>
        {getStatusBadge(assessment.status)}
      </div>

      <div className="space-y-2">
        <p className="text-gray-700 text-sm">{assessment.methodology}</p>
        {assessment.notes && (
          <p className="text-gray-600 text-sm italic">{assessment.notes}</p>
        )}
      </div>

      <div className="flex items-center justify-between pt-3 mt-3 border-t border-gray-200">
        <span className="text-gray-600 text-sm">Valor</span>
        <span className="text-red-600">R$ {assessment.price}</span>
      </div>
    </Card>
  );

  return (
    <div className="p-6 space-y-6">
      {/* Header */}
      <div>
        <h2 className="text-gray-900 text-2xl mb-2">Minhas Avaliações</h2>
        <p className="text-gray-600">Acompanhe seu histórico e progresso</p>
      </div>

      {/* Tabs */}
      <Tabs defaultValue="all" className="w-full">
        <TabsList className="grid w-full grid-cols-4 bg-gray-100">
          <TabsTrigger value="all" className="data-[state=active]:bg-red-600 data-[state=active]:text-white">
            Todas
          </TabsTrigger>
          <TabsTrigger value="requested" className="data-[state=active]:bg-red-600 data-[state=active]:text-white">
            Solicitadas
          </TabsTrigger>
          <TabsTrigger value="in-progress" className="data-[state=active]:bg-red-600 data-[state=active]:text-white">
            Em Andamento
          </TabsTrigger>
          <TabsTrigger value="completed" className="data-[state=active]:bg-red-600 data-[state=active]:text-white">
            Concluídas
          </TabsTrigger>
        </TabsList>

        <TabsContent value="all" className="space-y-4 mt-6">
          {mockAssessments.map((assessment) => (
            <AssessmentCard key={assessment.id} assessment={assessment} />
          ))}
        </TabsContent>

        <TabsContent value="requested" className="space-y-4 mt-6">
          {requestedAssessments.length > 0 ? (
            requestedAssessments.map((assessment) => (
              <AssessmentCard key={assessment.id} assessment={assessment} />
            ))
          ) : (
            <div className="text-center py-12">
              <p className="text-gray-500">Nenhuma avaliação solicitada</p>
            </div>
          )}
        </TabsContent>

        <TabsContent value="in-progress" className="space-y-4 mt-6">
          {inProgressAssessments.length > 0 ? (
            inProgressAssessments.map((assessment) => (
              <AssessmentCard key={assessment.id} assessment={assessment} />
            ))
          ) : (
            <div className="text-center py-12">
              <p className="text-gray-500">Nenhuma avaliação em andamento</p>
            </div>
          )}
        </TabsContent>

        <TabsContent value="completed" className="space-y-4 mt-6">
          {completedAssessments.length > 0 ? (
            completedAssessments.map((assessment) => (
              <AssessmentCard key={assessment.id} assessment={assessment} />
            ))
          ) : (
            <div className="text-center py-12">
              <p className="text-gray-500">Nenhuma avaliação concluída</p>
            </div>
          )}
        </TabsContent>
      </Tabs>

      {/* Assessment Detail Dialog */}
      <Dialog open={!!selectedAssessment} onOpenChange={() => setSelectedAssessment(null)}>
        <DialogContent className="max-w-[420px]">
          <DialogHeader>
            <DialogTitle>Detalhes da Avaliação</DialogTitle>
          </DialogHeader>
          {selectedAssessment && (
            <div className="space-y-4 py-4">
              <div className="flex items-center justify-between">
                <h3 className="text-gray-900">{selectedAssessment.professionalName}</h3>
                {getStatusBadge(selectedAssessment.status)}
              </div>

              <div className="space-y-3">
                <div className="flex justify-between text-sm">
                  <span className="text-gray-600">Data:</span>
                  <span className="text-gray-900">{formatDate(selectedAssessment.date)}</span>
                </div>
                <div className="flex justify-between text-sm">
                  <span className="text-gray-600">Metodologia:</span>
                  <span className="text-gray-900 text-right max-w-[200px]">
                    {selectedAssessment.methodology}
                  </span>
                </div>
                <div className="flex justify-between text-sm">
                  <span className="text-gray-600">Valor:</span>
                  <span className="text-red-600">R$ {selectedAssessment.price}</span>
                </div>
                {selectedAssessment.notes && (
                  <div>
                    <p className="text-gray-600 text-sm mb-1">Observações:</p>
                    <p className="text-gray-900 text-sm bg-gray-50 p-3 rounded-lg">
                      {selectedAssessment.notes}
                    </p>
                  </div>
                )}
              </div>

              {selectedAssessment.results && (
                <div className="pt-4 border-t border-gray-200">
                  <div className="flex items-center space-x-2 mb-4">
                    <Activity className="h-5 w-5 text-red-600" />
                    <h4 className="text-gray-900">Resultados</h4>
                  </div>
                  <div className="grid grid-cols-2 gap-3">
                    <div className="bg-red-50 p-3 rounded-lg">
                      <p className="text-gray-600 text-xs mb-1">Peso</p>
                      <p className="text-gray-900">{selectedAssessment.results.weight} kg</p>
                    </div>
                    <div className="bg-red-50 p-3 rounded-lg">
                      <p className="text-gray-600 text-xs mb-1">Altura</p>
                      <p className="text-gray-900">{selectedAssessment.results.height} cm</p>
                    </div>
                    <div className="bg-red-50 p-3 rounded-lg">
                      <p className="text-gray-600 text-xs mb-1">% Gordura</p>
                      <p className="text-gray-900">{selectedAssessment.results.bodyFat}%</p>
                    </div>
                    <div className="bg-red-50 p-3 rounded-lg">
                      <p className="text-gray-600 text-xs mb-1">Massa Muscular</p>
                      <p className="text-gray-900">{selectedAssessment.results.muscleMass} kg</p>
                    </div>
                    <div className="bg-red-50 p-3 rounded-lg col-span-2">
                      <p className="text-gray-600 text-xs mb-1">IMC</p>
                      <p className="text-gray-900">{selectedAssessment.results.imc}</p>
                    </div>
                  </div>
                </div>
              )}
            </div>
          )}
        </DialogContent>
      </Dialog>
    </div>
  );
}
